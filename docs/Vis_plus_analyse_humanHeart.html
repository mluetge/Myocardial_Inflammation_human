Idents(seuratSub) <- seuratSub$grp
grpVec <- unique(seuratSub$RNA_snn_res.0.25)

clustDE <- lapply(grpVec, function(grp){
  grpSub <- unique(seuratSub$RNA_snn_res.0.25)[which(
    unique(seuratSub$RNA_snn_res.0.25)==grp)]
  seuratSub2 <- subset(seuratSub, RNA_snn_res.0.25 == grpSub)
  DEgenes <-FindAllMarkers(seuratSub2, only.pos=T, logfc.threshold = 0.25,
                           min.pct = 0.1)
  if(nrow(DEgenes)>1){
    DEgenes <- DEgenes %>% filter(p_val_adj<0.01) %>%
      mutate(group=paste0(grp, "_", cluster)) %>% 
      mutate(geneID=gsub(".*\\.", "",  gene)) %>% 
      filter(nchar(gene)>1)
    }
})

names(clustDE) <- grpVec

clustDE_Dat <- data.frame(do.call("rbind", clustDE))
